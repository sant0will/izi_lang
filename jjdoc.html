// ======== FUNÇÕES =========== //

//Execucao do parser
Start ::= ( COMANDOS ) <EOF> 
//Função para todos os comandos possíveis
COMANDOS ::= ( WHILE | OUT | IN | IF | ID | FOR | EXPA | SWITCH | CASE | COMENTARIO | CALLF | FUNCTION | FOPEN )* 
//Função para comentarios
COMENTARIO ::= ( ( <COMENTARIO> ( <LITERALS> )* <COMENTARIO> ) ) 
//Funcoes para atribuicao
ID ::= ( <VAR> <TIPO> <LITERALS> ( "," IDR | <TO> IDIR )* <EOL> )
IDR ::= ( <LITERALS> | "," IDR )+
IDIR ::= ( <VALUES> | <TF> | <NUM> | <TEIA> ( "," <LITERALS> <TO> ( <VALUES> | <TF> | <NUM> | <TEIA> ) )* ) 
//Função para operações aritmeticas
EXPA ::= ( <COUNTS> <LITERALS> <TO> ( <VALUES> | <NUM> | <LITERALS> ) ( <OPA> ( <VALUES> | <NUM> | <LITERALS> ) )* <EOL> ) 
//Função para operações booleanas
EXPB ::= ( ( <VALUES> | <NUM> | <LITERALS> ) <OPB> ( <VALUES> | <NUM> | <LITERALS> ) )* 
//Funções para criação do for
FOR ::= ( <FOR> <PAR_A> ID EXPB <VIRG> EXPA <PAR_F> <CHA_A> ( COMANDOS ) <CHA_F> ) 
//Funções para a estrutura if
IFE ::= ( <PAR_A> EXPB <PAR_F> <CHA_A> ( COMANDOS ) <CHA_F> )*
IF ::= ( <IF> IFE ( <ELIF> IFE )* ( <ELSE> <CHA_A> ( COMANDOS ) <CHA_F> )* ) 
//Funções para entrada de dados
IN ::= ( <ECO> <LITERALS> <TO> <IN> <PAR_A> <TIPO> <PAR_F> <EOL> ) 
//Funções para entrada de dados
OUT ::= ( <AST> <OUT> <PAR_A> <LITERALS> <PAR_F> <EOL> ) 
//Funções da estrutura while
WHILE ::= ( <WHILE> <PAR_A> ( EXPB ) <PAR_F> <CHA_A> ( COMANDOS ) <CHA_F> ) 
//Funções da estrutura switch
CASE ::= <CASE> <VALUES> <CHA_A> ( COMANDOS ) <BREAK> <EOL> <CHA_F>
SWITCH ::= <SWITCH> <PAR_A> <LITERALS> <PAR_F> <CHA_A> ( COMANDOS ) <CHA_F> 
//Funções para funcoes
CALLF ::= ( <ARR> <FUNCTION> <DOT> <LITERALS> <PAR_A> <LITERALS> <PAR_F> <EOL> )
FUNCTION ::= ( <FUNCTION> <LITERALS> <PAR_A> <LITERALS> <PAR_F> <CHA_A> ( COMANDOS ) <CHA_F> ) 
//Funções para abertura de arquivos
FOPEN ::= ( <FOPEN> <PAR_A> <LITERALS> <PAR_F> <EOL> )
